# 数组求最值

定义max变量，for循环遍历数组中每一个值，如果当前数组索引大于max变量，替换max变量，最后得到就是最大值

int main()
{
	int arr[] = { -35,-5,-22,-44,-55 };
	int max = arr[0];

	int len = sizeof(arr) / sizeof(int);
	
	for (int i = 0;i < len;i++)
	{
		if (arr[i] > max)
		{
			max = arr[i];
		}
	}
	printf("%d\n", max);
	
	return 0;
}

生成10个随机数，且数值是1~100之间，求数组中所有数据的和

#include <stdio.h>
#include <stdlib.h>
#include <time.h>
int main()
{
	srand(time(NULL));

	int arr[10] = {0};
	int all = 0;
	
	int len = sizeof(arr) / sizeof(int);
	
	for (int i = 0;i < len;i++)
	{
		int num = rand() % 100 + 1;
		arr[i] = num;
		all += arr[i];
		printf("%d\n",arr[i]);
		
	}
	printf("%d\n", all);
	
	return 0;
}
